---
title: 'Create task'
description: 'Create a new task for the authenticated user.'
---

`POST /api/tasks`

Creates a new task owned by the authenticated user.

<Info>**Authentication:** Bearer access token required.</Info>

<Info>**Rate limit:** 50 requests per hour per user.</Info>

## Request body

<ParamField body="title" type="string" required>
  Task title. Maximum 200 characters.
</ParamField>

<ParamField body="description" type="string">
  Task description. Maximum 5000 characters.
</ParamField>

<ParamField body="status" type="string" default="pending">
  Initial task status. Allowed values: `pending`, `in_progress`, `completed`, `cancelled`.
</ParamField>

<ParamField body="priority" type="string" default="medium">
  Task priority. Allowed values: `low`, `medium`, `high`, `urgent`.
</ParamField>

<ParamField body="due_date" type="string">
  Due date in ISO 8601 format (e.g., `2024-12-31T23:59:59`).
</ParamField>

<ParamField body="tags" type="integer[]">
  Array of tag IDs to associate with the task.
</ParamField>

## Response fields

<ResponseField name="success" type="boolean">
  Whether the request succeeded.
</ResponseField>

<ResponseField name="message" type="string">
  Human-readable result message.
</ResponseField>

<ResponseField name="data" type="object">
  The created task object.
  <Expandable title="properties">
    <ResponseField name="id" type="integer">
      Unique task ID.
    </ResponseField>
    <ResponseField name="title" type="string">
      Task title.
    </ResponseField>
    <ResponseField name="description" type="string">
      Task description.
    </ResponseField>
    <ResponseField name="status" type="string">
      Task status.
    </ResponseField>
    <ResponseField name="priority" type="string">
      Task priority.
    </ResponseField>
    <ResponseField name="due_date" type="string">
      ISO 8601 due date, or `null` if not set.
    </ResponseField>
    <ResponseField name="completed_at" type="string">
      ISO 8601 completion timestamp, or `null`.
    </ResponseField>
    <ResponseField name="is_overdue" type="boolean">
      Whether the task is past its due date and not completed.
    </ResponseField>
    <ResponseField name="is_completed" type="boolean">
      Whether the task status is `completed`.
    </ResponseField>
    <ResponseField name="created_at" type="string">
      ISO 8601 creation timestamp.
    </ResponseField>
    <ResponseField name="updated_at" type="string">
      ISO 8601 last-updated timestamp.
    </ResponseField>
    <ResponseField name="user_id" type="integer">
      ID of the user who owns the task.
    </ResponseField>
    <ResponseField name="tags" type="object[]">
      Tags associated with the task.
      <Expandable title="tag properties">
        <ResponseField name="id" type="integer">
          Tag ID.
        </ResponseField>
        <ResponseField name="name" type="string">
          Tag name.
        </ResponseField>
        <ResponseField name="color" type="string">
          Tag color as a hex code.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
