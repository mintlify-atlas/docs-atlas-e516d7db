---
title: 'Task statistics'
description: 'Get a summary of task counts grouped by status and priority for the authenticated user.'
---

`GET /api/tasks/statistics`

Returns aggregated statistics for all tasks owned by the authenticated user, including counts by status, by priority, and overdue tasks.

<Info>**Authentication:** Bearer access token required.</Info>

## Response fields

<ResponseField name="success" type="boolean">
  Whether the request succeeded.
</ResponseField>

<ResponseField name="data" type="object">
  Task statistics for the current user.
  <Expandable title="properties">
    <ResponseField name="total_tasks" type="integer">
      Total number of tasks owned by the user.
    </ResponseField>
    <ResponseField name="completed_tasks" type="integer">
      Number of tasks with status `completed`.
    </ResponseField>
    <ResponseField name="pending_tasks" type="integer">
      Number of tasks with status `pending`.
    </ResponseField>
    <ResponseField name="in_progress_tasks" type="integer">
      Number of tasks with status `in_progress`.
    </ResponseField>
    <ResponseField name="cancelled_tasks" type="integer">
      Number of tasks with status `cancelled`.
    </ResponseField>
    <ResponseField name="overdue_tasks" type="integer">
      Number of tasks that are past their due date and not completed.
    </ResponseField>
    <ResponseField name="by_priority" type="object">
      Task counts grouped by priority level.
      <Expandable title="properties">
        <ResponseField name="low" type="integer">
          Number of tasks with priority `low`.
        </ResponseField>
        <ResponseField name="medium" type="integer">
          Number of tasks with priority `medium`.
        </ResponseField>
        <ResponseField name="high" type="integer">
          Number of tasks with priority `high`.
        </ResponseField>
        <ResponseField name="urgent" type="integer">
          Number of tasks with priority `urgent`.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
