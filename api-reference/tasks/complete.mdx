---
title: 'Complete task'
description: 'Mark a task as completed.'
---

`POST /api/tasks/{task_id}/complete`

Marks a task as `completed` and records the current time in `completed_at`. Regular users can only complete their own tasks. Admin users can complete any task.

<Info>**Authentication:** Bearer access token required.</Info>

## Path parameters

<ParamField path="task_id" type="integer" required>
  The ID of the task to mark as completed.
</ParamField>

## Response fields

<ResponseField name="success" type="boolean">
  Whether the request succeeded.
</ResponseField>

<ResponseField name="message" type="string">
  Human-readable result message.
</ResponseField>

<ResponseField name="data" type="object">
  The updated task object.
  <Expandable title="properties">
    <ResponseField name="id" type="integer">
      Unique task ID.
    </ResponseField>
    <ResponseField name="title" type="string">
      Task title.
    </ResponseField>
    <ResponseField name="description" type="string">
      Task description.
    </ResponseField>
    <ResponseField name="status" type="string">
      Task status. Will be `completed`.
    </ResponseField>
    <ResponseField name="priority" type="string">
      Task priority.
    </ResponseField>
    <ResponseField name="due_date" type="string">
      ISO 8601 due date, or `null`.
    </ResponseField>
    <ResponseField name="completed_at" type="string">
      ISO 8601 timestamp of when the task was marked completed.
    </ResponseField>
    <ResponseField name="is_overdue" type="boolean">
      Whether the task was past its due date (always `false` once completed).
    </ResponseField>
    <ResponseField name="is_completed" type="boolean">
      Whether the task status is `completed`. Will be `true`.
    </ResponseField>
    <ResponseField name="created_at" type="string">
      ISO 8601 creation timestamp.
    </ResponseField>
    <ResponseField name="updated_at" type="string">
      ISO 8601 last-updated timestamp.
    </ResponseField>
    <ResponseField name="user_id" type="integer">
      ID of the user who owns the task.
    </ResponseField>
    <ResponseField name="tags" type="object[]">
      Tags associated with the task.
      <Expandable title="tag properties">
        <ResponseField name="id" type="integer">
          Tag ID.
        </ResponseField>
        <ResponseField name="name" type="string">
          Tag name.
        </ResponseField>
        <ResponseField name="color" type="string">
          Tag color as a hex code.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
